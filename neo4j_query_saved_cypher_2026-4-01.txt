"task","query","id",
"restart database",
"// Delete all relationships
:auto MATCH ()-[r]->() CALL { WITH r DELETE r } IN TRANSACTIONS OF 50000 ROWS;

// Delete all nodes
:auto MATCH (n) CALL { WITH n DETACH DELETE n } IN TRANSACTIONS OF 50000 ROWS;",0,
"restart graph",
"// Drop stored graph to create new graph
CALL gds.graph.drop('tracksGraph');",1,
"load data",
"// Import csv and create track nodes with associated properties
LOAD CSV WITH HEADERS FROM 'file:///random_2000_regina_strokes.csv' AS row
CREATE (:Track {
    track_name: row.track_name,
    artists: row.artists,
    speechiness: toFloat(row.speechiness),
    acousticness: toFloat(row.acousticness),
    loudness: toFloat(row.loudness),
    energy: toFloat(row.energy),
    danceability: toFloat(row.danceability)
});",2,
"normalize",
"// Scale loudness to range [0-1] to match other numeric features
MATCH (t:Track)
SET
  t.norm_loudness = (t.loudness + 60) / 60;",3,
"make graph project",
"// Put all nodes and properties into a GDS graph project
CALL gds.graph.project(
  'tracksGraph',
  'Track',
  '*',
  {
    nodeProperties: ['speechiness', 'acousticness', 'norm_loudness', 'energy', 'danceability']
  }
);",4,
"makes knn",
"// Run k nearest neighbors on the following track properties
CALL gds.knn.write('tracksGraph', {
  nodeProperties: ['speechiness', 'acousticness', 'norm_loudness', 'energy', 'danceability'],
  topK: 20,
  similarityCutoff: 0.9, // similarity match of 0.9 or higher needed
  writerelationshipType: 'SIMILAR',
  writeProperty: 'score'
});",5,
"make recs",
"WITH ['Regina Spektor', 'The Strokes'] AS artists

// Find top 5 rec nodes
MATCH (t:Track)
WHERE t.artists IN artists

MATCH (t)-[:SIMILAR]->(rec:Track)
WHERE NOT rec.artists IN artists

WITH artists, rec, count(*) AS score
ORDER BY score DESC
LIMIT 5

// Then recompute scores ONLY from those nodes
WITH artists, collect(rec) AS topRecs

MATCH (t:Track)
WHERE t.artists IN artists

MATCH (t)-[:SIMILAR]->(rec)
WHERE rec IN topRecs

RETURN rec.track_name, rec.artists, count(*) AS score
ORDER BY score DESC;",6,
"recs visualization",
"// Start the same as previous query to find top 5 rec nodes
WITH ['Regina Spektor', 'The Strokes'] AS artists

MATCH (t:Track)
WHERE t.artists IN artists

MATCH (t)-[:SIMILAR]->(rec:Track)
WHERE NOT rec.artists IN artists

WITH artists, rec, count(*) AS score
ORDER BY score DESC
LIMIT 5

// Collect the exact nodes
WITH artists, collect(rec) AS topRecs

// Then ONLY use those nodes
MATCH (t:Track)
WHERE t.artists IN artists

MATCH (t)-[r:SIMILAR]->(rec)
WHERE rec IN topRecs

RETURN t, r, rec;",7
